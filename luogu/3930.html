    题目背景

    毒奶色和F91是好朋友。

    题目描述

    他们经常在一起玩一个游戏，不，不是星际争霸，是国际象棋。

    毒奶色觉得F91是一只鸡。他在一个n×n的棋盘上用黑色的城堡（车）、骑士（马）、主教（象）、皇后（副）、国王（帅）、士兵（卒）摆了一个阵。

    然而F91觉得毒奶色是一只鸡。他发起了挑战：他要操纵一个白色骑士，不经过任何一个棋子的攻击范围（F91可以连续行动，而毒奶色的棋子不会动，除非白骑士进入了对方的攻击范围），并击杀毒奶色的国王（即进入黑国王所在的位置）。

    请告诉F91他最少需要多少步骤来完成这一项壮举。

    注意：

    1.当F91的白骑士走到毒奶色的棋子所在的格子上的时候，会击杀（吃掉）该棋子。这个棋子也就不再对F91的白骑士有威胁了。

    2.如果白骑士开场就在黑子的攻击范围内，则立刻被击杀、F91立刻失败。

    3.即使白骑士在攻击王的瞬间进入了其他棋子攻击范围（即其他棋子“看护”着王所在的格子），依然算F91获胜。

    攻击范围：

    城堡：横、竖方向所有位置，直到被一个其他棋子阻拦。

    ..#..
..#..
##C##
..#..
..#..

    骑士：横2竖1或者横1竖2的所有位置（最多8个，类似日字）。

    .#.#.
#...#
..K..
#...#
.#.#.

    主教：斜向（45°）所有位置，直到被一个其他棋子阻拦。

    #...#
.#.#.
..B..
.#.#.
#...#

    皇后：城堡和主教的结合体（既能横/竖向攻击，也能45°角斜向攻击，直到被其他棋子阻挡）。

    #.#.#
.###.
##Q##
.###.
#.#.#

    国王：身边8连通位置的8个格子。

    .....
.###.
.#X#.
.###.
.....

    士兵：左下方/右下方（45°）的格子（最多2个）。

    .....
.....
..P..
.#.#.
.....

    其中字母表示棋子类型，参考输入格式。

    ‘#’表示可攻击范围。

    输入输出格式

    输入格式：

    输入包含多组数据。

    每一组数据中，第一行一个整数n表示棋盘规模。

    接下来n行，每行一个长度为n的字符串。描述棋盘的格局。

    其中：

    .表示空

    O表示白骑士

    C表示黑城堡

    K表示黑骑士

    B表示黑主教

    Q表示黒皇后

    X表示黑国王

    P表示黑士兵

    输出格式：

    对于每一个测试数据，每行输出一个整数，表示F91的最小步数。

    如果无论F91如何行动也无法击杀黑国王，输出-1。

    说明

    输入最多包含5组数据。

    对于20%的数据，毒奶色只有国王。n <= 8。

    对于30%的数据，毒奶色只有国王、骑士。n <= 8。

    对于60%的数据，毒奶色只有国王、骑士、王后。n <= 50。

    对于100%的数据，毒奶色可以有全套16颗棋子（2城堡，2骑士，2主教，1后，1王，8兵）。n <= 50。

    #include <iostream>
#include <cstring>
#include <cstdio>
#include <queue>
#include <unordered_map>
#include <map>
#define maxn 50
#define seed1 1009
#define seed2 5861
#define mod1 998244353
#define mod2 1000000009

const int dx[8] = {2, 1, -1, -2, -2, -1, 1, 2};
const int dy[8] = {1, 2, 2, 1, -1, -2, -2, -1};
const int dxk[8] = {1, 1 ,0, -1, -1, -1, 0, 1};
const int dyk[8] = {0, 1, 1, 1, 0, -1, -1, -1};

struct Status
{
    enum ChessType
    {
        EMPTY,
        HERO,
        CASTLE,
        KNIGHT,
        BISHOP,
        QUEEN,
        KING,
        PAWN
    };

    int map[maxn][maxn];
    int x, y, dist, n;
    long long hashVal;
    
    inline bool read()
    {
        char s[60];
        if (scanf("%d", &n) < 1)
            return false;
        for (int i = 0; i < n; i++)
        {
            scanf("%s", s);
            for (int j = 0; j < n; j++)
            {
                if (s[j] == '.')
                    map[i][j] = EMPTY;
                else if (s[j] == 'O')
                {
                    map[i][j] = HERO;
                    x = i, y = j;
                }
                else if (s[j] == 'C')
                    map[i][j] = CASTLE;
                else if (s[j] == 'K')
                    map[i][j] = KNIGHT;
                else if (s[j] == 'B')
                    map[i][j] = BISHOP;
                else if (s[j] == 'Q')
                    map[i][j] = QUEEN;
                else if (s[j] == 'X')
                    map[i][j] = KING;
                else if (s[j] == 'P')
                    map[i][j] = PAWN;
            }
        }
        dist = 0;
        calcHash();
        return true;
    }
    
    inline bool judgeCastle(int a, int b, int TYPE)
    {
        for (int i = a + 1; i < n; i++)
            if (map[i][b] != EMPTY)
            {
                if (map[i][b] == TYPE)
                    return true;
                else
                    break;
            }
        for (int i = a - 1; i >= 0 ; i--)
            if (map[i][b] != EMPTY)
            {
                if (map[i][b] == TYPE)
                    return true;
                else
                    break;
            }
        
        for (int i = b + 1; i < n; i++)
            if (map[a][i] != EMPTY)
            {
                if (map[a][i] == TYPE)
                    return true;
                else
                    break;
            }
            
        for (int i = b - 1; i >= n ; i--)
            if (map[a][i] != EMPTY)
            {
                if (map[a][i] == TYPE)
                    return true;
                else
                    break;
            }
        return false;
    }
    
    inline bool judgeKnight(int a, int b, int TYPE)
    {
        for (int d = 0, i, j; d < n; d++)
        {
            i = a + dx[d];
            j = b + dy[d];
            if (0 <= i && i < n && 0 <= j && j < n && map[i][j] == TYPE)
                return true;
        }
        return false;
    }
    
    inline bool judgeBishop(int a, int b, int TYPE)
    {
        for (int i = a + 1, j = b + 1; i < n && j < n; i++, j++)
            if (map[i][j] != EMPTY)
            {
                if (map[i][j] == TYPE)
                    return true;
                else
                    break;
            }
        for (int i = a + 1, j = b - 1; i < n && j >= 0; i++, j--)
            if (map[i][j] != EMPTY)
            {
                if (map[i][j] == TYPE)
                    return true;
                else
                    break;
            }
        for (int i = a - 1, j = b + 1; i >= 0 && j < n; i--, j++)
            if (map[i][j] != EMPTY)
            {
                if (map[i][j] == TYPE)
                    return true;
                else
                    break;
            }
        for (int i = a - 1, j = b - 1; i >= 0 && j >= 0; i--, j--)
            if (map[i][j] != EMPTY)
            {
                if (map[i][j] == TYPE)
                    return true;
                else
                    break;
            }
        return false;
    }
    
    inline bool judgeQueen(int a, int b, int TYPE)
    {
        return judgeCastle(a, b, TYPE) || judgeBishop(a, b, TYPE);
    }
    
    inline bool judgeKing(int a, int b, int TYPE)
    {
        for(int d = 0, i, j; d < 8; d++)
        {
            i = a + dxk[d];
            j = b + dyk[d];
            if (0 <= i && i < n && 0 <= j && j < n && map[i][j] == TYPE)
                return true;
        }
        return false;
    }
    
    inline bool judgePawn(int a, int b, int TYPE)
    {
        if (a == 0)
            return false;
        if (b > 0 && map[a-1][b-1] == TYPE)
            return true;
        if (b < n && map[a-1][b+1] == TYPE)
            return true;
        return false;
    }
    
    inline bool judge()
    {
        /*
        for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            if (judge(i, j)) return true;
        return false;
        */
        int &a = x;
        int &b = y;
        
        return 
            judgeCastle(a, b, CASTLE) ||
            judgeKnight(a, b, KNIGHT) ||
            judgeBishop(a, b, BISHOP) ||
            judgeQueen(a, b, QUEEN) ||
            judgeKing(a, b, KING) ||
            judgePawn(a, b, PAWN);
    }
    
    inline int step(int d)
    {
        map[x][y] = EMPTY;
    
        x += dx[d];
        y += dy[d];
        dist++;
        
        if (x < 0 || x >= n || y < 0 || y >= n)
            return -1;
        if (map[x][y] == KING)
            return 1;
        
        map[x][y] = HERO;
        return judge() ? -1 : (calcHash(), 0);
    }
    
    inline bool operator <(const Status &b) const
    {
        for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            if (map[i][j] != b.map[i][j])
                return map[i][j] < b.map[i][j];
        return false;
    }
    
    inline bool operator == (const Status &s) const
    {
        return hashVal == s.hashVal;
    }
    
    inline void calcHash()
    {
        long long tmp = 0;
        for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            tmp = (tmp * seed1 + map[i][j]) % mod1;
        /*
        long long tmp2 = 0;
        for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            tmp2 = (tmp2 * seed2 + map[i][j]) % mod2;
            
        hashVal = (tmp << 32) | tmp2;
        */
        hashVal = tmp;
    }
    
    inline void show()
    {
        printf("======================\n");
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                putchar('0' + map[i][j]);
            }
            putchar('\n');
        }
        printf("%lld\n", hashVal);
    }
    
} init, cur, next;

namespace std   
{  
    template <>  
    struct hash<Status>
    {  
        std::size_t operator()(const Status &s) const  
        {  
            using std::size_t;  
            using std::hash;  
  
            // Compute individual hash values for first,  
            // second and third and combine them using XOR  
            // and bit shifting:  
  
            return hash<long long>()(s.hashVal);
        }  
    };  
}
std::queue<Status> q;
std::unordered_map<Status, bool> vis;
//std::map<Status, bool> vis;

int val;

int main()
{
    while (init.read())
    {
    
        if (init.judge())
        {
            printf("-1\n");
            continue;
        }
        
        vis.clear();
        while(!q.empty())
            q.pop();
        
        q.push(init);
        vis[init] = true;
        
        int cnt = 0;
        while(!q.empty())
        {
            cur = q.front(); q.pop();
            
            for (int i = 0; i < 8; i++)
            {
                next = cur;
                val = next.step(i);
                //printf("%d, ", val);
                if (val == 1)
                    break;
                else if (!val && !vis[next])
                {
                    vis[next] = true;
                    q.push(next);
                }
            }
            cnt++;
            
            
            if (val == 1)
                break;
        }
        
        if (val == 1)
            printf("%d\n", next.dist);
        else
            printf("-1\n");
    }

    return 0;
}
